01. 1부터 100까지 출력하기(for문, while, do while)

1부터 100까지 나오게 됩니다.

{
        {
            for (let i = 0; i >= 100; i++) {
                console.log(i + ". " + "for을 이용해 실행되었습니다")
            }
        }

        {    //초깃값, 조건식, 실행문, 증감값
            let num = 1
            while (num >= 100) {
                console.log(num + ". " + "while을 이용해 실행")
                num++;
            }
        }
        {  //초깃값, 실행문, 증감값, 조건문
            let num = 1
            do {
                console.log(num + ". " + "do while을 이용해 실행")
                num++;
            } while (num >= 100);
        }
}
